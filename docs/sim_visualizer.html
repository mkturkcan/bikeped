sim_visualizer

Scenario visualizer for the bikeped decision testbench.

Renders simulated fisheye camera views of test scenarios showing what the camera would actually see: agents projected through the fisheye model onto a ground-plane wireframe, with decision state overlay.

Camera placement matches index.html presets:

  • Single pole: (x=5, y=-13, h=5.5), heading 90 deg (facing +y across road)
  • Cross-cam: add second camera at (x=5, y=+13, h=5.5), heading 270 deg
Usage:

python sim_visualizer.py # key scenarios, single cam python sim_visualizer.py --scenarios 1 5 7 # specific scenarios python sim_visualizer.py --two-cam # two-camera split view python sim_visualizer.py --no-save # display only

 57class FisheyeRenderer:
 58    """Thin wrapper around FisheyeCamera for rendering at arbitrary resolution.
 59
 60    Delegates all projection math to FisheyeCamera (decision_testbench.py)
 61    so that the simulation visualiser uses the exact same model as the
 62    decision testbench and error analysis.
 63
 64    The calibrated 3500x3500 camera is scaled to the render resolution
 65    (default 640x480) by adjusting radius_px and optical center proportionally.
 66    """
 67
 68    def __init__(self, cam_x, cam_y, cam_h, heading_deg, fov_deg=197.90,
 69                 img_w=640, img_h=480, pitch_deg=None,
 70                 cx_offset=0.0, cy_offset=0.0):
 71        self.img_w = img_w
 72        self.img_h = img_h
 73
 74        # Scale calibrated parameters from 3500x3500 to render resolution
 75        calib_size = 3500
 76        render_dim = min(img_w, img_h)
 77        scale = render_dim / calib_size
 78        scaled_radius = int(round(1750 * scale))
 79
 80        # Optical center in render coords
 81        render_cx = (img_w - 1) / 2.0 + cx_offset
 82        render_cy = (img_h - 1) / 2.0 + cy_offset
 83
 84        self._cam = FisheyeCamera(
 85            height=cam_h,
 86            fov_deg=fov_deg,
 87            radius_px=scaled_radius,
 88            frame_width=img_w,
 89            frame_height=img_h,
 90            kappa=1.0,
 91            cam_x=cam_x,
 92            cam_y=cam_y,
 93            pitch_deg=pitch_deg,
 94            heading_deg=heading_deg,
 95            cx_override=render_cx,
 96            cy_override=render_cy,
 97        )
 98
 99        # Sensor visible band scaled to render resolution
100        # In the 3500x3500 crop: rows 670-2829 are on the sensor
101        self.sensor_v_min = int(round(self._cam.sensor_v_min * scale))
102        self.sensor_v_max = int(round(self._cam.sensor_v_max * scale))
103
104    def project(self, world_points):
105        """Project world points (N,3) to pixel coordinates (N,2).
106        Returns pixels and a valid mask."""
107        pts = np.atleast_2d(world_points).astype(np.float64)
108        pixels, valid = self._cam.world_to_pixel(pts)
109        return pixels, valid
110
111    def project_single(self, x, y, z):
112        """Project a single world point. Returns (u, v, valid)."""
113        pts, valid = self.project(np.array([[x, y, z]]))
114        return int(pts[0, 0]), int(pts[0, 1]), bool(valid[0])

117def draw_ground_grid(frame, renderer, spacing=2.0, extent=50.0):
118    """Draw a ground-plane grid through the fisheye projection."""
119    color = (50, 50, 50)
120    step = 0.5  # finer interpolation for smooth curves under fisheye
121    # Lines along x
122    for y in np.arange(-extent, extent + spacing, spacing):
123        pts_3d = np.array([[x, y, 0.0] for x in np.arange(-extent, extent + step, step)])
124        pts_2d, valid = renderer.project(pts_3d)
125        for i in range(len(pts_2d) - 1):
126            if valid[i] and valid[i+1]:
127                p1 = (int(pts_2d[i, 0]), int(pts_2d[i, 1]))
128                p2 = (int(pts_2d[i+1, 0]), int(pts_2d[i+1, 1]))
129                cv2.line(frame, p1, p2, color, 1)
130    # Lines along y
131    for x in np.arange(-extent, extent + spacing, spacing):

141def draw_crosswalk(frame, renderer):
142    """Draw crosswalk stripes at x=5, spanning y=-12 to y=12."""
143    cw_x = 5.0
144    for stripe_y in np.arange(-12, 12, 2.0):

153            poly = pts_2d.astype(np.int32).reshape(-1, 1, 2)

157def draw_agent_3d(frame, renderer, agent, trail):
158    """Draw an agent as a 3D wireframe box projected through fisheye,
159    oriented along the agent's heading direction."""
160    from decision_testbench import OBJECT_TYPES, object_3d_corners
161
162    x, y = agent['x'], agent['y']
163    heading = agent.get('heading', 0.0)
164    is_person = agent['type'] == 'person'
165
166    if is_person:
167        obj = OBJECT_TYPES['pedestrian']
168        color = (80, 80, 255)
169        label = f"P{agent['id']}"
170    else:
171        obj = OBJECT_TYPES['cyclist']
172        color = (80, 255, 80)
173        label = f"B{agent['id']}"
174
175    corners = object_3d_corners(x, y, obj, heading)
176    pts_2d, valid = renderer.project(corners)
177
178    if not np.any(valid):
179        return
180
181    edges = [(0,1),(1,2),(2,3),(3,0),  # bottom
182             (4,5),(5,6),(6,7),(7,4),  # top
183             (0,4),(1,5),(2,6),(3,7)]  # verticals
184
185    for i, j in edges:
186        if valid[i] and valid[j]:
187            p1 = (int(pts_2d[i, 0]), int(pts_2d[i, 1]))
188            p2 = (int(pts_2d[j, 0]), int(pts_2d[j, 1]))
189            cv2.line(frame, p1, p2, color, 1, cv2.LINE_AA)
190
191    # Label at top center
192    top_center = np.array([[x, y, obj.height + 0.3]])
193    lbl_px, lbl_valid = renderer.project(top_center)
194    if lbl_valid[0]:
195        lp = (int(lbl_px[0, 0]) + 5, int(lbl_px[0, 1]))
196        cv2.putText(frame, label, lp,
197                    cv2.FONT_HERSHEY_SIMPLEX, 0.4, color, 1)
198
199    # Trail on ground
200    if len(trail) >= 2:
201        trail_pts = np.array([[p[0], p[1], 0.0] for p in trail[-30:]])
202        t_2d, t_valid = renderer.project(trail_pts)
203        for i in range(len(t_2d) - 1):
204            if t_valid[i] and t_valid[i+1]:
205                p1 = (int(t_2d[i, 0]), int(t_2d[i, 1]))
206                p2 = (int(t_2d[i+1, 0]), int(t_2d[i+1, 1]))
207                cv2.line(frame, p1, p2, (color[0]//3, color[1]//3, color[2]//3),
208                         1, cv2.LINE_AA)

261def draw_bev_radar(agents, trail_history, cam_x, cam_y, heading_deg, state,
262                   bev_w=200, bev_h=200, max_range=25.0):
263    """Draw a bird's-eye-view radar showing agent positions relative to camera.
264
265    The camera's forward direction (heading) points up in the BEV.
266    This matches mqtt_bridge's BEV where the camera is at the origin
267    and forward = +x (displayed as up).
268    """
269    bev = np.full((bev_h, bev_w, 3), 30, dtype=np.uint8)
270
271    # Camera at bottom center
272    cx = bev_w // 2
273    cy = int(bev_h * 0.85)
274    scale = bev_h * 0.75 / max_range
275
276    # Heading rotation: transform world offset to camera-relative (fwd, right)
277    h_rad = heading_deg * pi / 180.0
278    cos_h, sin_h = cos(h_rad), sin(h_rad)
279
280    def world_to_bev(wx, wy):
281        """World position -> BEV pixel. Forward = up, right = right."""
282        dx = wx - cam_x
283        dy = wy - cam_y
284        # Rotate to camera frame: fwd = along heading, right = perpendicular
285        fwd = dx * cos_h + dy * sin_h
286        right = dx * sin_h - dy * cos_h
287        bx = int(cx + right * scale)
288        by = int(cy - fwd * scale)
289        return bx, by
290
291    # Range rings
292    for r_m in [5, 10, 15, 20, 25]:
293        r_px = int(r_m * scale)
294        cv2.circle(bev, (cx, cy), r_px, (50, 50, 50), 1)
295        if r_m % 10 == 0:
296            label_y = cy - r_px
297            if 5 < label_y < bev_h - 5:
298                cv2.putText(bev, f"{r_m}m", (cx + 3, label_y + 10),
299                            cv2.FONT_HERSHEY_SIMPLEX, 0.25, (80, 80, 80), 1)
300
301    # Camera marker
302    cv2.circle(bev, (cx, cy), 3, (255, 255, 255), -1)
303
304    # Agents
305    for a in agents:
306        bev_x, bev_y = world_to_bev(a['x'], a['y'])
307
308        if not (0 <= bev_x < bev_w and 0 <= bev_y < bev_h):
309            continue
310
311        is_person = a['type'] == 'person'
312        color = (80, 80, 255) if is_person else (80, 255, 80)
313
314        # Trail
315        trail = trail_history.get(a['id'], [])
316        if len(trail) >= 2:
317            for i in range(max(0, len(trail) - 20), len(trail) - 1):
318                tx1, ty1 = world_to_bev(trail[i][0], trail[i][1])
319                tx2, ty2 = world_to_bev(trail[i+1][0], trail[i+1][1])
320                if (0 <= tx1 < bev_w and 0 <= ty1 < bev_h and
321                    0 <= tx2 < bev_w and 0 <= ty2 < bev_h):
322                    cv2.line(bev, (tx1, ty1), (tx2, ty2),
323                             (color[0]//3, color[1]//3, color[2]//3), 1)
324
325        cv2.circle(bev, (bev_x, bev_y), 4, color, -1)
326        label = f"P{a['id']}" if is_person else f"B{a['id']}"
327        cv2.putText(bev, label, (bev_x + 5, bev_y - 3),
328                    cv2.FONT_HERSHEY_SIMPLEX, 0.25, color, 1)
329
330    # State indicator
331    state_color = STATE_COLORS.get(state, (128, 128, 128))
332    cv2.rectangle(bev, (2, 2), (bev_w - 3, 12), state_color, -1)
333    cv2.putText(bev, STATE_LABELS.get(state, state.upper()), (5, 10),
334                cv2.FONT_HERSHEY_SIMPLEX, 0.28, (255, 255, 255), 1)
335
336    return bev

377def render_camera_frame(renderer, agents, trail_history, prox_max,
378                        scenario_name, t, duration, state, danger,
379                        cam_label="", cam_x=0.0, cam_y=0.0,
380                        heading_deg=90.0, error_data=None):
381    """Render one frame from a single camera's perspective with BEV radar.
382
383    If error_data is provided (list of dicts with true_x/y, obs_x/y, err_m),
384    draws error vectors from true to observed positions.
385    """
386    frame = np.full((IMG_H, IMG_W, 3), 20, dtype=np.uint8)
387
388    draw_ground_grid(frame, renderer, spacing=2.0, extent=25.0)
389    draw_crosswalk(frame, renderer)
390
391    # Proximity lines
392    persons = [a for a in agents if a['type'] == 'person']
393    bikes = [a for a in agents if a['type'] == 'bike']
394    for b in bikes:
395        for p in persons:
396            d = sqrt((b['x'] - p['x'])**2 + (b['y'] - p['y'])**2)
397            if d < prox_max:
398                draw_proximity_line(frame, renderer, b['x'], b['y'], p['x'], p['y'])
399
400    # Agents (true positions)
401    for a in agents:
402        trail = trail_history.get(a['id'], [])
403        draw_agent_3d(frame, renderer, a, trail)
404
405    # Error vectors (true → observed)
406    if error_data:
407        for ed in error_data:
408            draw_error_vector(frame, renderer,
409                              ed['true_x'], ed['true_y'],
410                              ed['obs_x'], ed['obs_y'], ed['err_m'])
411
412    # Sensor boundary (dims regions outside the physical 3840x2160 sensor)
413    draw_sensor_boundary(frame, renderer)
414
415    draw_hud(frame, scenario_name, t, duration, state, danger, cam_label)
416
417    # BEV radar overlay (bottom-right)
418    bev_w, bev_h = 180, 180
419    bev = draw_bev_radar(agents, trail_history, cam_x, cam_y, heading_deg,
420                         state, bev_w, bev_h)
421    margin = 6
422    y1 = IMG_H - bev_h - margin
423    x1 = IMG_W - bev_w - margin
424    roi = frame[y1:y1+bev_h, x1:x1+bev_w]
425    cv2.addWeighted(bev, 0.85, roi, 0.15, 0, dst=roi)
426    frame[y1:y1+bev_h, x1:x1+bev_w] = roi
427
428    return frame

431def render_scenario(scenario, scenario_idx, params_name='optimized',
